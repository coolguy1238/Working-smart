  <!-- SPA Core Runtime Router Script -->
  <script>
    const TITLES = {
      "home": "ValueAI – AI that thinks with you",
      "chat": "ValueAI – Chat Engine",
      "pricing": "ValueAI – Pricing Tiers",
      "signup": "ValueAI – Provision Account",
      "login": "ValueAI – Log In",
      "company": "ValueAI – Core Mission",
      "research": "ValueAI – Frontier Research"
    };
    let currentPage = null;

    function navigate(page, anchor) {
      if(currentPage === page && !anchor) return;
      currentPage = page;

      // Toggle View Active Classes
      document.querySelectorAll('.spa-view').forEach(v => v.classList.remove('active'));
      const activeView = document.getElementById('view-' + page);
      if(activeView) activeView.classList.add('active');

      // Update Nav Link States
      document.querySelectorAll('.nav-lnk-item').forEach(link => link.classList.remove('active'));
      const activeLink = document.getElementById('lnk-' + page);
      if(activeLink) activeLink.classList.add('active');

      // Visibility Rules for Shell Layout Elements
      const navBar = document.getElementById('global-nav');
      const footerBar = document.getElementById('global-footer');
      if (page === 'chat' || page === 'login' || page === 'signup') {
        if(footerBar) footerBar.style.display = 'none';
        if((page === 'login' || page === 'signup') && navBar) navBar.style.display = 'none';
      } else {
        if(navBar) navBar.style.display = 'flex';
        if(footerBar) footerBar.style.display = 'block';
      }

      document.title = TITLES[page] || 'ValueAI';
      history.pushState({page}, '', '#' + page);
      window.scrollTo(0, 0);
      closeDrw();
      injectHamburgers();
    }

    window.addEventListener('popstate', e => {
      navigate((e.state && e.state.page) || 'home');
    });

    // Boot Initialization Sequence
    (function() {
      const h = location.hash.replace('#','');
      const valid = ["home", "chat", "pricing", "signup", "login", "company", "research"];
      navigate(valid.includes(h) ? h : 'home');
    })();

    // Drawer Interface Controllers
    function openDrw()  { document.getElementById('mob-drawer').classList.add('open'); }
    function closeDrw() { document.getElementById('mob-drawer').classList.remove('open'); }

    function injectHamburgers() {
      const nav = document.getElementById('global-nav');
      if(nav && !nav.querySelector('.mob-ham')) {
        const btn = document.createElement('button');
        btn.className = 'mob-ham';
        btn.onclick = openDrw;
        btn.innerHTML = '<span></span><span></span><span></span>';
        nav.appendChild(btn);
      }
    }

    // Dynamic Live Simulation Functional Chat Block
    function sendMessage() {
      const input = document.getElementById('chat-input');
      const container = document.getElementById('chat-messages');
      const txt = input.value.trim();
      if(!txt) return;

      // User Message Render Box
      container.innerHTML += `<div class="msg-wrap"><div class="bubble user">${txt}</div></div>`;
      input.value = '';
      container.scrollTop = container.scrollHeight;

      // Simulated Intelligent Response Generation
      setTimeout(() => {
        container.innerHTML += `<div class="msg-wrap"><div class="bubble ai">Processing verification request pipeline matrices for prompt concept: "${txt}". Operations completed successfully.</div></div>`;
        container.scrollTop = container.scrollHeight;
      }, 750);
    }

    // Reset Chat Simulation Function
    function resetChat() {
      document.getElementById('chat-messages').innerHTML = `<div class="msg-wrap"><div class="bubble ai">Welcome to ValueAI. How can I assist your engineering workspace or technical pipelines today?</div></div>`;
    }
  </script>
